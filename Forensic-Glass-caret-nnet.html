function (x, y, len = NULL, search = "grid") 
{
    if (search == "grid") {
        out <- expand.grid(size = ((1:len) * 2) - 1, decay = c(0, 
            10^seq(-1, -4, length = len - 1)))
    }
    else {
        out <- data.frame(size = sample(1:20, size = len, replace = TRUE), 
            decay = 10^runif(len, min = -5, 1))
    }
    out
}
